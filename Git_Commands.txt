

git init - Иницализирует репозиторий в текущей папке. По умолчанию создаст ветку мастер

git status - Показывает статус репозитория

git add - Добавляет новые файлы к проекту

git add [file_name] - Добавляет файл [file_name]

git add . - Добавляет все файлы в директории

git mv [file_name0] [file_name1] - Переименовывает [file_name0] в [file_name1]

git reset - Отменяет индексирование

git commit - Фиксирует изменения в репозитории. флаг -m позволит задать сообщение фиксации

git rm - Удаляет файлы и добавляет сведения об их удалении в Git. Флаг -f принудительно удалит файл и из рабочей директории и из индекса

Чтобы удалить файл только из индекса воспользуемся --cached [file_name]

git amend - Изменяет последний коммит

git checkout, git revert - Отменяют изменения

git clone - Клонирует проект с удалённого репозитория

git branch — Показывает список веток

git branch branch-name — Создает новую ветку branch-name

git config - Конфигурация и параметры git

git config --global user.name — Показывает имя пользователя
git config --global user.name 'User_Name' — Изменяет имя пользователя
git config --global user.email — Показывает email пользователя
git config --global user.email 'User@mail.ru' — Изменяет email пользователя

git push - Проталкивает текущие локальные коммиты в удаленный репозиторий

git pull - Забирает изменения с удаленного репозитория в локальный
